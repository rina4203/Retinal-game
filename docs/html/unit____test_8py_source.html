42def mock_rect_side_effect(*args, **kwargs):
43 """
44 @brief Custom side-effect for mocking Pygame Rectangles.
45 @details Allows the mock object to store x, y, w, h attributes and behave like a real Rect.
46 It sets `colliderect` to return False by default, which can be overridden in tests.
47 @param args Arguments passed to Rect constructor (x, y, w, h).
48 @return A MagicMock object with Rect attributes.
49 """
50 rect = MagicMock()
51 # Default behavior: No collision
52 rect.colliderect.return_value = False
53 rect.x = args[0] if len(args) > 0 else 0
54 rect.y = args[1] if len(args) > 1 else 0
55 rect.w = args[2] if len(args) > 2 else 0
56 rect.h = args[3] if len(args) > 3 else 0
57 rect.width = rect.w
58 rect.height = rect.h
59 rect.left = rect.x
60 rect.right = rect.x + rect.w
61 rect.top = rect.y
62 rect.bottom = rect.y + rect.h
63 rect.centerx = rect.x + rect.w // 2
64 rect.centery = rect.y + rect.h // 2
65 rect.collidepoint.return_value = False
66 return rect
67
68# Apply the side effect to the Rect mock
69mock_pygame.Rect = MagicMock(side_effect=mock_rect_side_effect)
