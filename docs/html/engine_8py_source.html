23def measure_time(func):
24 @functools.wraps(func)
25 def wrapper(*args, **kwargs):
26 start_time = time.perf_counter()
27 result = func(*args, **kwargs)
28 end_time = time.perf_counter()
29 execution_time = (end_time - start_time) * 1000
30 logging.info(f"Performance: {func.__name__} took {execution_time:.4f} ms")
